<div id="content-header">
  <h1>JCS_DestroySpawnEffect</h1>
</div>

<p>
  Effect that spawn a gameobject after this gameobject is destroyed.
</p>


<br/>
<h2>Variables</h2>
<br/>

<table>
  <tr>
    <td>mSpawnCount</td>
    <td>How many transform spawn.</td>
  </tr>
  <tr>
    <td>mActiveWhatever</td>
    <td>Active this effect by what ever this object is destoryed.</td>
  </tr>
  <tr>
    <td>mActiveWithHitList</td>
    <td>Active the effect by the destroy time.</td>
  </tr>
  <tr>
    <td>mActiveWithDestroyTime</td>
    <td>Active the effect by the destroy time.</td>
  </tr>
  <tr>
    <td>mSamePosition</td>
    <td>Play the animation as the same position as the destroyed gameobject.</td>
  </tr>
  <tr>
    <td>mSameRotation</td>
    <td>Play the animation as the same rotation as the destroyed gameobject.</td>
  </tr>
  <tr>
    <td>mSameScale</td>
    <td>Play the animation as the same scale as the destroyed gameobject.</td>
  </tr>
  <tr>
    <td>mRandPos</td>
    <td>Randomize the position when spawn the gameobject.</td>
  </tr>
  <tr>
    <td>mRandPosRange</td>
    <td>Random position value added.</td>
  </tr>
  <tr>
    <td>mRandRot</td>
    <td>Randomize the rotation when spawn the gameobject.</td>
  </tr>
  <tr>
    <td>mRandRotRange</td>
    <td>Random rotation value added.</td>
  </tr>
  <tr>
    <td>mRandScale</td>
    <td>Randomize the scale when spawn the gameobject.</td>
  </tr>
  <tr>
    <td>mRandScaleRange</td>
    <td>Random scale value added.</td>
  </tr>
</table>
